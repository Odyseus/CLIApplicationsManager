timers = {}  # type: Dict[Key, threading.Timer]



[docs]def debounce(callback, delay, key):
    # type: (Callable[[], None], float, Key) -> threading.Timer
    try:
        timers[key].cancel()
    except KeyError:
        pass

    timers[key] = timer = threading.Timer(delay / 1000, callback)
    timer.start()
    return timer
